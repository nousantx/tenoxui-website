# Color Class

This is just a simple tutorial to make a bunch of className using `makeStyles` function to generate classNames for color and background from `css variable`'s value.

## Let's Start!

You can use this project tree example :

```txt title="Project Tree"
root
├── src
│   ├── App.jsx
│   ├── Color.js
│   ├── index.css
│   └── main.jsx
├── package.json
└── // other
```
First, you need to define each color you need inside your CSS file, and domt forget to add it to `main.jsx` file :

```css title="index.css"
:root {
  /* Primary: Gǎn Lǎn Huáng Olive */
  --primary-100: #f9fff2;
  --primary-200: #e4ffc1;
  --primary-300: #d4ff90;
  --primary-400: #cbff5f;
  --primary-500: #c7ff2e;
  --primary-600: #a2c90f;
  --primary-700: #7d9303;
  --primary-800: #545c00;
  --primary-900: #252600;

  /* Accent: Blue Blue */
  --accent-100: #f2f2ff;
  --accent-200: #bdc1fe;
  --accent-300: #8592f8;
  --accent-400: #4c67e8;
  --accent-500: #1842c9;
  --accent-600: #0835a1;
  --accent-700: #022b78;
  --accent-800: #00204f;
  --accent-900: #001226;
}
```

After that, you can make a function to generate color classNames and then use `makeStyles` function to generate the usable classNames :

```js title="Color.js"
import { makeStyles } from "tenoxui";

export default function Color(categories) {
  // store the color object
  const color = {};

  categories.forEach((category) => {
    for (let i = 100; i <= 900; i += 100) {
      // suffix shorthand
      const clsSuffix = `-${category}-${i}`;
      const valSuffix = `-[${category}-${i}]`;

      // add more class and style needed
      color[`.text${clsSuffix}`] = `tc${valSuffix}`;
      color[`.bg${clsSuffix}`] = `bg${valSuffix}`;
    }
  });

  // now, just apply the color
  makeStyles(color);
}
```

And finally, you can call the function inside your `App.jsx` file :

```jsx title="App.jsx"
import { useLayoutEffect } from "react";
import tenoxui from "tenoxui";
import Color from "./Color.js";

const App = () => {
  useLayoutEffect(() => {
    // add more color needed
    Color(["primary", "accent"]);
    // tenoxui init
    tenoxui();
  }, []);

  return (
    <div className="box-100px bg-accent-800 p-2rem br-0.125rem">
      <h1 className="text-primary-500">Hello</h1>
    </div>
  );
};

export default App;
```
